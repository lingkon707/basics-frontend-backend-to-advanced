# Git commands to check anything on Windows PC

# 1. Check Git version
git --version

# 2. Check current repository status-------
git status

# 3. Check current branch--------------
git branch
git branch --show-current

# 4. Check remote repositories-----------
git remote -v

# 5. Show commit log------------
git log
git log --oneline
git log --graph --all --decorate    --------

# 6. Show detailed commit info
git show <commit-hash>

# 7. Check differences (diff)
git diff
git diff --staged

# 8. Check configuration settings----------
git config --list
git config user.name
git config user.email

# 9. Check all branches and their latest commits --------
git branch -v

# 10. Check stashes
git stash list

# 11. Check remote branches
git branch -r

# 12. Check repository size ---------- 
git count-objects -vH

# 13. Check last commit on each branch
git for-each-ref --sort=-committerdate refs/heads/ --format='%(refname:short) - %(committerdate:relative) - %(authorname)'

# 14. Check file history
git log -- <filename>


# Bonus: Check Git help
git help
git <command> --help


# folder and files ---------
tree /F
git ls-files


# Create a new branch
git branch <branch-name>

# Switch branch
git checkout <branch-name>
# or (modern)
git switch <branch-name>

# Create and switch branch
git checkout -b <branch-name>
# or
git switch -c <branch-name>

# Delete a branch
git branch -d <branch-name>
git branch -D <branch-name>  # Force delete

--------------------------------------------------------
# Undo last commit but keep changes staged
git reset --soft HEAD~1

# Undo last commit and unstage changes
git reset HEAD~1

# Hard reset to discard changes completely (dangerous)
git reset --hard HEAD~1
